87def BellmanFord(times, time_limit): 

88 """ 

89 Bellman-Ford algorithm using all vertices as source. 

90 We pass in the time_limit to check, when there is a negative cycle, 

91 if it it reachable from the start. 

92 """ 

93 # Initialization 

94 n = len(times) 

95 d = [n*[float("inf")] for _ in range(n)] 

96 for s in range(n): 

97 d[s][s] = 0 

98 # Relaxation 

99 for i in range(n-1): 

100 updated = False 

101 for u in range(n): 

102 for v in range(n): 

103 distReach = d[s][u] + times[u][v] 

104 if d[s][v] > d[s][u] + times[u][v]: 

105 updated = True 

106 d[s][v] = distReach 

107 # An optimization. It doesn't change the worst case, but well ... 

108 if not updated: 

109 break 

110 # Negative cycle detection 

111 for u in range(n): 

112 for v in range(n): 

113 if d[s][v] > d[s][u] + times[u][v] and d[0][u] < time_limit: 

114 return True, d # Negative cycle 

115 return False, d # no negative cycles
